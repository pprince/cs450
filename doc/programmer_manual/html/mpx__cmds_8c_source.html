00083                                  * Print out the 'ambiguous command' header,

00085                                 printf("Ambiguous command: %s\n", name);
00086                                 printf("    Matches:\n");

00088                                 printf("        %s\n", this_command->name);

00090                                 /* This is a subsequent duplicate match;

00092                                  * so we only need to print out the current command name. */
00093                                 printf("        %s\n", this_command->name);

00095                 }
00096 
00097                 this_command = this_command->next;
00098         }
00099 
00100         /* If we got a command name that matches unambiguously, run that command. */
00101         if ( num_matches == 1 ){
00102                 first_match->function(argc, argv);

00104         
00105         /* Otherwise, if we got no matches at all, say so. */
