00467 /*
00468 
00469         Procedure: sys_alloc_mem
00470 
00471         Purpose: Allocate a memory block
00472 
00473         Parameters:
00474         
00475                 size_t size       No. of bytes to allocate
00476 
00477         Returns: void* pointer to allocated block;
00478                  null pointer in case of error
00479                  
00480         Calls: calloc
00481         
00482         Globals: alloc_table, alloc_ix, num_alloc
00483 
00484 
00485         For program loading (Module R-4), the blocks must be aligned.
00486         This is achieved by requesting one extra paragraph and
00487         returning the lowest aligned address as the base.  It is
00488         necessary to keep a table of allocated address, original and
00489         aligned, to support correct freeing.
00490 
00491         It would seem simpler to use the Turbo-C function allocmem,
00492         which always allocates aligned blocks.  However, allocmem
00493         appears to conflict with the internal allocation of fopen
00494         and other C library routines.
00495 
00496 
00497 */
00498 
00499 void *sys_alloc_mem (      size_t   size     /* size in bytes to allocate */
00500                     )
00501 {
00502         int ix_save;      /* temp copy of table index */
00503         void *addr;        /* addr returned by calloc (*void) */
00504         word offset;      /* offset of unaligned address */
00505         word seg;         /* segment addr of unaligned address */
00506         void *addr_alig; /* aligned address */
00507         int rem; /* temp for alignment computation */
00508 
00509         /* ensure that allocation table is not full */
00510         if (num_alloc >= MAX_ALLOC) {
00511                 return(NULL);
00512         }
00513 
00514         /* find free entry in allocation table */
00515         ix_save = alloc_ix;
00516         while (alloc_table[alloc_ix].original != NULL) {
00517                 alloc_ix++;
00518                 if (alloc_ix >= MAX_ALLOC) alloc_ix = 0;
00519 
00520                 /* safety check -- should never happen! */
00521                 if (alloc_ix == ix_save) return(NULL);
00522         }
00523 
00524         /* call allocation routine */
00525         /* request 15 extra bytes to ensure alignment is possible */
00526         addr = calloc(size + 15,1);
00527 
00528         /* compute aligned base */
00529         offset = FP_OFF(addr);
00530         seg = FP_SEG(addr);
00531         rem = offset % 16;
00532         if (rem > 0) offset = offset + 16 - rem;
00533         addr_alig = MK_FP(seg,offset);
00534 
00535         /* insert entries in table */
00536         alloc_table[alloc_ix].original = addr;
00537         alloc_table[alloc_ix].aligned = addr_alig;
00538 
00539 
00540         /* increment count */
00541         num_alloc++;
00542 
00543         return(addr_alig);
00544 
00545 }

00571 int sys_free_mem (         void     *ptr     /* pointer to block */
00572                  )
00573 {
00574         void *free_addr;  /* true (unaligned) block address */
00575         int free_ix;               /* temp table index */
00576 
00577         /* ensure valid pointer */
00578         if (ptr==NULL) return(ERR_SUP_INVMEM);
00579 
00580         /* Look for the block in the allocation table */
00581         free_addr = NULL;
00582         for (free_ix=0; free_ix<MAX_ALLOC; free_ix++) {
00583                 if (alloc_table[free_ix].aligned == ptr) {
00584                         free_addr = alloc_table[free_ix].original;
00585                         break;
00586                 }
00587         }
00588 
00589         /* If the block wasn't found, report error */
00590         if (free_addr == NULL) return(ERR_SUP_INVMEM);
00591 
00592 
00593         /* free the block & clear the table entry */
00594         alloc_table[free_ix].original = NULL;
00595         alloc_table[free_ix].aligned = NULL;
00596         free(free_addr);
00597 
00598         /* decrement count */
00599         num_alloc--;
00600 
00601         return(OK);
00602

00687                  )
00688 {
00689         char temp_path[MAX_PATH_SIZE];
00690         int res;
00691         int len;
00692 
00693         /* ensure there is no directory currently open */
00694         if (current_path[0]!=NULCH) return(ERR_SUP_DIROPN);
00695 
00696         /* check pathname for validity */
00697 
00698         /* construct temporary search path name */
00699         temp_path[0] = NULCH;
00700         if (path_name != NULL) {
00701                 strncpy(temp_path,path_name,MAX_PATH_SIZE-4);
00702         }
00703 
00704         /* build wildcard search string */
00705         len = strlen(temp_path);
00706 
00707         /* if path is null, use current directory */
00708         if (len==0) strcpy(temp_path,"*.*");
00709 
00710         /* handle paths ending in backslash, including root */
00711         else if (temp_path[len-1]=='\\') strcat(temp_path,"*.*");
00712 
00713         /* handle paths not ending in backslash */
00714         else strcat(temp_path,"\\*.*");
00715 
00716         /* validate the directory by searching for either an ordinary
00717            file or a directory in it.  This should never fail; even
00718            empty directories contain the directories "." and ".."
00719         */
00720         
00721         res = findfirst(temp_path, &file_block, FA_DIREC);
00722         if (res!=0)  return(ERR_SUP_INVDIR);
00723 
00724         /* save pathname as current, clear entry count */
00725         current_path[0] = NULCH;
00726         if (path_name != NULL) {
00727                 strncpy(current_path,path_name,MAX_PATH_SIZE-6);
00728         }
00729         num_entries = 0;
00730 
00731         return(OK);
